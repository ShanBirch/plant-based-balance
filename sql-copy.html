-- 1. Create the game_matches table
CREATE TABLE IF NOT EXISTS public.game_matches (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    challenger_id UUID REFERENCES public.users(id) ON DELETE CASCADE,
    opponent_id UUID REFERENCES public.users(id) ON DELETE CASCADE,
    game_type TEXT NOT NULL,
    coin_bet INTEGER DEFAULT 0,
    current_turn UUID REFERENCES public.users(id),
    game_state JSONB DEFAULT '{"board": null}'::jsonb,
    status TEXT DEFAULT 'pending' CHECK (status IN ('pending', 'active', 'completed', 'draw', 'forfeit')),
    winner_id UUID REFERENCES public.users(id),
    move_count INTEGER DEFAULT 0,
    last_move_at TIMESTAMPTZ DEFAULT NOW(),
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 2. Create the game_moves table to track board states
CREATE TABLE IF NOT EXISTS public.game_moves (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    match_id UUID REFERENCES public.game_matches(id) ON DELETE CASCADE,
    player_id UUID REFERENCES public.users(id) ON DELETE CASCADE,
    move_number INTEGER NOT NULL,
    move_data JSONB NOT NULL,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 3. Enable Row Level Security (RLS)
ALTER TABLE public.game_matches ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.game_moves ENABLE ROW LEVEL SECURITY;

-- 4. Setup Security Policies for game_matches
CREATE POLICY "Users can view their own game matches"
    ON public.game_matches FOR SELECT
    USING (auth.uid() = challenger_id OR auth.uid() = opponent_id);

CREATE POLICY "Users can create game matches"
    ON public.game_matches FOR INSERT

CREATE POLICY "Users can update their own game matches"
    ON public.game_matches FOR UPDATE
    USING (auth.uid() = challenger_id OR auth.uid() = opponent_id);
